drop table if exists actors;
drop table if exists directors;
drop table if exists cst;
drop table if exists movies;
drop table if exists people;




--people table, ties all people together


Create table People (
          PID                 char(4) NOT NULL,
        fName                text,
        lName                 text,
        address        text,
spouseName         text,
          PRIMARY KEY (PID)
);


-- actors table 
Create table actors (
        PID                 char(4) NOT NULL references people(pid),
        birthDate         date,
        hairColor         text,
        eyeColor         text,
        height                 integer,
        weight                integer,
        favColor        text,
        SAGADate        date,
          PRIMARY KEY (PID)
);


-- Movies Table
Create table Movies (
name                text,
releaseYear        int,
MPAA                int NOT NULL,
DBOS                text,
FBOS                text,
        bluRaySales        text,
        PRIMARY KEY (MPAA)
);


--Directors Table
Create table Directors (




PID                         char(4) NOT NULL references People(PID),
filmSchoolAttended         text,
DGADate                 date,
favoriteLensMaker         text,
PRIMARY KEY (PID)
);




--Cast Table
Create table Cst (
PID                 char(4) NOT NULL  references People(PID),
MPAA                int NOT NULL references Movies(MPAA),
role                 text,
PRIMARY KEY (PID, role)


);

--Select directors with whom Roger Moore has worked


SELECT people.fName, people.lName
FROM people 
INNER JOIN Cst ON people.pid = Cst.pid
WHERE role = 'Director'
AND Cst.MPAA IN ( 
        SELECT        Cst.mpaa
        FROM Cst
        INNER JOIN People ON People.PID = Cst.PID
        WHERE People.fName = 'Roger'
        AND People.lName = 'Moore'
 );
